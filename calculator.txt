*&---------------------------------------------------------------------*
*& Report  ZZARYAN_CALC
*&
*&---------------------------------------------------------------------*
*&
*&
*&---------------------------------------------------------------------*
report zzaryan_calc.

data: txt1       type p decimals 2,
      txt2       type p decimals 2,
      txt_result type p decimals 2.


call screen 0300.
*&---------------------------------------------------------------------*
*&      Module  ADD_0300  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
module add_0300 input.
  if sy-ucomm = 'BADD'.
    TXT_RESULT = txt1 + txt2.
  endif.
  IF SY-UCOMM = 'BACK'.
    LEAVE TO SCREEN 0.
  ENDIF.
endmodule.
*&---------------------------------------------------------------------*
*&      Module  SUB_0300  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
module sub_0300 input.
  if sy-ucomm = 'BSUB'.
    txt_result = txt1 - txt2.
  endif.
endmodule.
*&---------------------------------------------------------------------*
*&      Module  MUL_0300  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
module mul_0300 input.
  if sy-ucomm = 'BMUL'.
    txt_result = txt1 * txt2.
  endif.
endmodule.
*&---------------------------------------------------------------------*
*&      Module  STATUS_0300  OUTPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
module status_0300 output.
  set pf-status 'ZCAL'.
*  SET TITLEBAR 'xxx'.
endmodule.
